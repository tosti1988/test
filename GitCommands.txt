#Git commands 
git status #checks current status of repository
git add -A #adds altered and untracked files to staging area
git commit -am "<message>" #commit changes with message
git push #push committed changes to Github 
git pull #pull any changes from Github dwon to local repo


git log #hows you the history of commits 
git log -3 #last three --oneline #in only one line # like this(git log -3 --oneline)
